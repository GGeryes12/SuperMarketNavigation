def non_dominated_sorting(fitness):
    fronts = [[]]
    domination_counts = [0] * len(fitness)
    dominated_solutions = [[] for _ in range(len(fitness))]
    for p in range(len(fitness)):
        for q in range(len(fitness)):
            if all(fitness[p] <= fitness[q]) and any(fitness[p] < fitness[q]):
                dominated_solutions[p].append(q)
            elif all(fitness[q] <= fitness[p]) and any(fitness[q] < fitness[p]):
                domination_counts[p] += 1
        if domination_counts[p] == 0:
            fronts[0].append(p)
    i = 0
    while len(fronts[i]) > 0:
        next_front = []
        for p in fronts[i]:
            for q in dominated_solutions[p]:
                domination_counts[q] -= 1
                if domination_counts[q] == 0:
                    next_front.append(q)
        fronts.append(next_front)
        i += 1
    return fronts[:-1]

    /// <summary>
    /// Provides methods for non-dominated sorting and diversity calculations.
    /// </summary>
    public static class NonDominatedSorting
    {
        /// <summary>
        /// Performs non-dominated sorting on a population and divides it into Pareto fronts.
        /// </summary>
        /// <param name="population">Population to sort.</param>
        /// <returns>A list of Pareto fronts, where each front contains a list of individuals.</returns>
        public static List<List<Individual>> Perform(Population population)
        {
            var fronts = new List<List<Individual>>();
            var dominatedSolutions = new Dictionary<Individual, List<Individual>>();
            var dominationCount = new Dictionary<Individual, int>();

            foreach (var p in population.Individuals)
            {
                dominatedSolutions[p] = new List<Individual>();
                dominationCount[p] = 0;

                foreach (var q in population.Individuals)
                {
                    if (Dominates(p, q))
                        dominatedSolutions[p].Add(q);
                    else if (Dominates(q, p))
                        dominationCount[p]++;
                }

                if (dominationCount[p] == 0)
                {
                    p.Rank = 1;
                    if (fronts.Count == 0) fronts.Add(new List<Individual>());
                    fronts[0].Add(p);
                }
            }

            int i = 0;
            while (fronts.Count > i && fronts[i].Count > 0)
            {
                var nextFront = new List<Individual>();
                foreach (var p in fronts[i])
                {
                    foreach (var q in dominatedSolutions[p])
                    {
                        dominationCount[q]--;
                        if (dominationCount[q] == 0)
                        {
                            q.Rank = i + 2;
                            nextFront.Add(q);
                        }
                    }
                }
                if (nextFront.Count > 0) fronts.Add(nextFront);
                i++;
            }

            return fronts;
        }

        /// <summary>
        /// Determines if one individual dominates another.
        /// </summary>
        /// <param name="p">First individual.</param>
        /// <param name="q">Second individual.</param>
        /// <returns>True if <paramref name="p"/> dominates <paramref name="q"/>; otherwise, false.</returns>
        private static bool Dominates(Individual p, Individual q)
        {
            bool atLeastOneBetter = false;
            for (int i = 0; i < p.Objectives.Length; i++)
            {
                if (p.Objectives[i] > q.Objectives[i]) return false;
                if (p.Objectives[i] < q.Objectives[i]) atLeastOneBetter = true;
            }
            return atLeastOneBetter;
        }

public static Individual Order1Crossover(Individual parent1, Individual parent2, Random random)
{
    int length = parent1.Path.Count;
    int start = random.Next(0, length);
    int end = random.Next(start, length);

    // Create child with empty placeholders
    var childPath = new List<string>(new string[length]);

    // Copy segment from Parent 1
    for (int i = start; i <= end; i++)
    {
        childPath[i] = parent1.Path[i];
    }

    // Fill remaining positions from Parent 2
    int currentIndex = 0;
    for (int i = 0; i < length; i++)
    {
        if (!childPath.Contains(parent2.Path[i]))
        {
            // Skip over filled positions in the child
            while (childPath[currentIndex] != null)
            {
                currentIndex++;
            }
            childPath[currentIndex] = parent2.Path[i];
        }
    }

    return new Individual(childPath);
}

Bresenham’s Line Algorithm (Adapted for Grid Traversal)

This algorithm works by incrementally stepping through the grid and checking which cells are crossed.

Helper Function to Compute Intersected Cells

csharp
using System;
using System.Collections.Generic;

public static class GridTraversal
{
    /// <summary>
    /// Computes the grid cells (aisles) that a line passes through.
    /// </summary>
    /// <param name="grid">2D grid representing the supermarket layout.</param>
    /// <param name="start">Start position (row, column).</param>
    /// <param name="end">End position (row, column).</param>
    /// <returns>A list of aisles (or grid cells) the line passes through.</returns>
    public static List<string> GetIntersectedAisles(string[,] grid, (int row, int col) start, (int row, int col) end)
    {
        var intersectedAisles = new List<string>();

        int x0 = start.col, y0 = start.row;
        int x1 = end.col, y1 = end.row;

        int dx = Math.Abs(x1 - x0);
        int dy = Math.Abs(y1 - y0);

        int sx = x0 < x1 ? 1 : -1;
        int sy = y0 < y1 ? 1 : -1;

        int err = dx - dy;

        while (true)
        {
            // Add the current cell to the intersected aisles list
            if (!string.IsNullOrEmpty(grid[y0, x0]) && grid[y0, x0] != "0") // Avoid empty spaces
            {
                if (!intersectedAisles.Contains(grid[y0, x0]))
                {
                    intersectedAisles.Add(grid[y0, x0]);
                }
            }

            // Check if we've reached the end point
            if (x0 == x1 && y0 == y1) break;

            // Update error and step
            int e2 = 2 * err;
            if (e2 > -dy)
            {
                err -= dy;
                x0 += sx;
            }
            if (e2 < dx)
            {
                err += dx;
                y0 += sy;
            }
        }

        return intersectedAisles;
    }
}
